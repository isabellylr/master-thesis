	\tikzstyle{neuron}=[circle,draw=black,,minimum size=12pt,inner sep=0pt,text width=2em, text centered]

	\node[neuron] (N-2) at (0,-3) {};
	\node[neuron] (N-3) at (3,0) {};
	\node[neuron] (N-4) at (3,-3) {};

	\path[] (N-2) edge [loop left] node {} (N-2);

}
